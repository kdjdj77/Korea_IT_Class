switch(표현식)
{
	case 값1:
		표현식==값1 일 때 실행되는 영역
		break;
	case 값2:
		표현식==값2 일 때 실행되는 영역
		break;
	...
	default:
		표현식이 case에 해당되지 않을 때
		break;
}

반복문
	특정 코드가 반복될 때 사용한다
for문
	for ( int i = 0; i < max; i++ )
	{
		반복할 코드;
	}
	for문을 만나면 최초 한 번 초기식을 실행한다
	그 다음으로 조건식을 판단한다
	조건식이 참이라면 for문 안쪽의 문장을 실행한다

	for문 닫는 중괄호를 만나면 증감식을 실행한다
	그 뒤로 조건식을 판단한다
	조건식이 참이라면 for문 안쪽의 문장을 실행한다
	만약 조건식이 거짓이라면 더 이상 for문 안쪽으로 들어가지
	않고 for문 다음 문장을 실행한다(반복종료)

복합대입연산자
	num += 값 -->	num = num + 값
	num -= 값  -->	num = num - 값
	num *= 값  -->	num = num * 값
	num /= 값  -->	num = num / 값
	num %= 값 -->	num = num % 값
continue
	반복문 안에서 사용되며, 자신을 감싸는 가장 가까운 반복의
	즉시 다음 반복을 실행항다
break
	반복문 안에서 사용되면, 자신을 감싸는 가장 가까운 반복을
	중단한다
